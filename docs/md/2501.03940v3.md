2 """
3 Returns the logits of the model for the given texts.
4 """
5 hs, metrics, attention_mask = self.pretrained_model.forward(texts) # [B, L, H], [B, L-1, M], [B, L]
6 attention_mask = attention_mask[:, :-1] # [B, L-1]
7 B, L, H = hs.size()
8
9 # Apply metrics_nn to obtain metrics features
10 processed_metrics = self.metrics_nn(metrics) # [B, L-1, F]
11
12 # Create semantic and positional embeddings
13 pos_embed = torch.arange(L-1, device=hs.device).unsqueeze(0).expand(B, -1) # [B, L-1]
14 pos_embed = pos_embed.unsqueeze(-1) / self.max_len # [B, L-1, 1]
15 gate_x_list = [hs[..., :-1, :], hs[..., 1:, :], pos_embed]
16 gate_x = torch.cat(gate_x_list, dim=-1)
17
18 # Dropout a percentage of tokens for regularization
19 gate_mask = self.__dropout_tokens(attention_mask == 0) # [B, L-1]
20
21 # Apply gate_nn to obtain gate logits, and make sure to mask padding and dropped tokens
22 gate_logits = self.gate_nn(gate_x) # [B, L-1, G], where G divides F
23 gate_logits = gate_logits.masked_fill(gate_mask.unsqueeze(-1), float("-inf")) # [B, L-1, G]
24
25 # If the number of gates is less than the number of features, repeat the gate logits
26 G, F = gate_logits.size(-1), processed_metrics.size(-1)
27 if 1 < G < F: gate_logits = gate_logits.repeat(1, 1, F // G) # [B, L-1, F]
28
29 # Apply the softmax across seq len to obtain weights
30 gates = gate_logits.softmax(dim=-2) # [B, L-1, F]
31 # Apply the attention mechanism to obtain the weighted sum of the features
32 aggregate_output = gates * processed_metrics # [B, L-1, F]
33 aggregate_output = aggregate_output.sum(dim=-2) # [B, F]
34
35 # Apply the final MLP to obtain the logits
36 logits = self.aggregate_nn(aggregate_output) # [B, 1]
37 return logits.squeeze(-1)
